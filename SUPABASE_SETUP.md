# Supabase setup for Device Testing

Supabase is a better fit for this frontend than direct MSSQL because the React app can use the Supabase JavaScript client with a public anon key. The important rule is: enable Row Level Security and add policies before exposing tables to the browser.

Official docs:

- Supabase tables: https://supabase.com/docs/guides/database/tables
- Supabase Row Level Security: https://supabase.com/docs/guides/database/postgres/row-level-security
- Supabase JavaScript client: https://supabase.com/docs/reference/javascript

## 1. Create a Supabase project

1. Go to https://supabase.com
2. Create a new project.
3. Open Project Settings -> API.
4. Copy:
   - Project URL
   - anon public key

## 2. Add environment variables

Create `.env.local` in this project:

```env
VITE_SUPABASE_URL=https://your-project.supabase.co
VITE_SUPABASE_ANON_KEY=your-anon-key
```

For Vercel, add the same variables:

Project Settings -> Environment Variables

Use:

- `VITE_SUPABASE_URL`
- `VITE_SUPABASE_ANON_KEY`

## 3. Install Supabase client

```bash
npm install @supabase/supabase-js
```

## 4. Create database tables

Run this in Supabase SQL Editor.

```sql
create table public.clients (
  id bigint generated by default as identity primary key,
  name text not null,
  client_code text not null unique,
  is_active boolean not null default true,
  created_at timestamptz not null default now()
);

create table public.statuses (
  id bigint generated by default as identity primary key,
  name text not null unique,
  color text not null default '#4b5563',
  is_active boolean not null default true,
  created_at timestamptz not null default now()
);

create table public.inventory_items (
  id bigint generated by default as identity primary key,
  date_delivered date,
  serial_no_cloud_id text not null,
  device_model text,
  client_id bigint references public.clients(id),
  status_id bigint not null references public.statuses(id),
  remarks text,
  date_deployed date,
  created_at timestamptz not null default now(),
  updated_at timestamptz
);

create table public.device_inventory_items (
  id bigint generated by default as identity primary key,
  company text,
  client_id bigint references public.clients(id) on delete set null,
  raised_by text,
  date_received date,
  package_style text,
  cst_number text,
  ticket_number text,
  sn_number text,
  device_type text,
  with_adapter text,
  start_repairing_support date,
  end_date_support date,
  start_qa date,
  end_date_qa date,
  status_id bigint references public.statuses(id) on delete set null,
  date_delivered date,
  give_to text,
  remarks text,
  created_at timestamptz not null default now(),
  updated_at timestamptz
);

insert into public.clients (name, client_code)

on conflict (name) do nothing;
```

## 5. Enable Row Level Security

For development, this allows logged-in users to manage rows. Later, you can make policies stricter.

```sql
alter table public.clients enable row level security;
alter table public.statuses enable row level security;
alter table public.inventory_items enable row level security;
alter table public.device_inventory_items enable row level security;

create policy "authenticated users can read clients"
on public.clients for select
to authenticated
using (true);

create policy "authenticated users can manage clients"
on public.clients for all
to authenticated
using (true)
with check (true);

create policy "authenticated users can read statuses"
on public.statuses for select
to authenticated
using (true);

create policy "authenticated users can manage statuses"
on public.statuses for all
to authenticated
using (true)
with check (true);

create policy "authenticated users can read inventory"
on public.inventory_items for select
to authenticated
using (true);

create policy "authenticated users can manage inventory"
on public.inventory_items for all
to authenticated
using (true)
with check (true);

create policy "authenticated users can read device inventory"
on public.device_inventory_items for select
to authenticated
using (true);

create policy "authenticated users can manage device inventory"
on public.device_inventory_items for all
to authenticated
using (true)
with check (true);
```

If you want the app usable without login during early testing, replace `to authenticated` with `to anon, authenticated`. Do not use that for production unless the data is truly public.

For quick testing without login, run this policy block instead:

```sql
drop policy if exists "public read clients" on public.clients;
drop policy if exists "public manage clients" on public.clients;
drop policy if exists "public read statuses" on public.statuses;
drop policy if exists "public manage statuses" on public.statuses;
drop policy if exists "public read device inventory" on public.device_inventory_items;
drop policy if exists "public manage device inventory" on public.device_inventory_items;

create policy "public read clients"
on public.clients for select
to anon, authenticated
using (true);

create policy "public manage clients"
on public.clients for all
to anon, authenticated
using (true)
with check (true);

create policy "public read statuses"
on public.statuses for select
to anon, authenticated
using (true);

create policy "public manage statuses"
on public.statuses for all
to anon, authenticated
using (true)
with check (true);

create policy "public read device inventory"
on public.device_inventory_items for select
to anon, authenticated
using (true);

create policy "public manage device inventory"
on public.device_inventory_items for all
to anon, authenticated
using (true)
with check (true);
```

## 6. Create Supabase client file

Create `src/lib/supabase.js`:

```js
import { createClient } from "@supabase/supabase-js";

export const supabase = createClient(
  import.meta.env.VITE_SUPABASE_URL,
  import.meta.env.VITE_SUPABASE_ANON_KEY
);
```

## 7. Query examples

Fetch dropdown data:

```js
const { data, error } = await supabase
  .from("clients")
  .select("id, name, is_active")
  .order("name");
```

Fetch inventory with joined client/status:

```js
const { data, error } = await supabase
  .from("inventory_items")
  .select(`
    id,
    date_delivered,
    serial_no_cloud_id,
    device_model,
    remarks,
    date_deployed,
    clients ( id, name ),
    statuses ( id, name, color )
  `)
  .order("id", { ascending: true });
```

Insert inventory:

```js
const { error } = await supabase.from("inventory_items").insert({
  date_delivered: form.dateDelivered || null,
  serial_no_cloud_id: form.serialNoCloudId,
  device_model: form.deviceModel || null,
  client_id: form.clientId || null,
  status_id: form.statusId,
  remarks: form.remarks || null,
  date_deployed: form.dateDeployed || null,
});
```

Update inventory:

```js
const { error } = await supabase
  .from("inventory_items")
  .update({
    serial_no_cloud_id: form.serialNoCloudId,
    updated_at: new Date().toISOString(),
  })
  .eq("id", id);
```

Delete inventory:

```js
const { error } = await supabase
  .from("inventory_items")
  .delete()
  .eq("id", id);
```
