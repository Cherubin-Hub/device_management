# Ongoing Testing Supabase SQL

Paste this into Supabase SQL Editor and run it.

```sql
create table if not exists public.ongoing_testing_items (
  id bigint generated by default as identity primary key,
  client_id bigint references public.clients(id) on delete set null,
  date_received date,
  package_style text,
  picture_url text,
  model text,
  with_adapter text,
  serial_number text,
  status_id bigint references public.statuses(id) on delete set null,
  repair_by text,
  test_by text,
  senior_test_by text,
  remarks text,
  source_inventory_id bigint,
  created_at timestamptz not null default now(),
  updated_at timestamptz
);

alter table public.ongoing_testing_items enable row level security;

alter table public.ongoing_testing_items
add column if not exists client_id bigint references public.clients(id) on delete set null;

create index if not exists ongoing_testing_items_client_id_idx
on public.ongoing_testing_items(client_id);

drop policy if exists "public read ongoing testing" on public.ongoing_testing_items;
drop policy if exists "public manage ongoing testing" on public.ongoing_testing_items;

create policy "public read ongoing testing"
on public.ongoing_testing_items
for select
to anon, authenticated
using (true);

create policy "public manage ongoing testing"
on public.ongoing_testing_items
for all
to anon, authenticated
using (true)
with check (true);

insert into storage.buckets (id, name, public)
values ('ongoing-testing-images', 'ongoing-testing-images', true)
on conflict (id) do update set public = true;

drop policy if exists "public read ongoing testing images" on storage.objects;
drop policy if exists "public upload ongoing testing images" on storage.objects;

create policy "public read ongoing testing images"
on storage.objects
for select
to anon, authenticated
using (bucket_id = 'ongoing-testing-images');

create policy "public upload ongoing testing images"
on storage.objects
for insert
to anon, authenticated
with check (bucket_id = 'ongoing-testing-images');
```

For production, replace `to anon, authenticated` with `to authenticated` and add login.
