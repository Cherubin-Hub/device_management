# Device Inventory Supabase SQL

Paste this into Supabase SQL Editor and run it.

```sql
create table if not exists public.device_inventory_items (
  id bigint generated by default as identity primary key,
  company text,
  client_id bigint references public.clients(id) on delete set null,
  raised_by text,
  date_received date,
  package_style text,
  cst_number text,
  ticket_number text,
  sn_number text,
  device_type text,
  with_adapter text,
  start_repairing_support date,
  end_date_support date,
  start_qa date,
  end_date_qa date,
  status_id bigint references public.statuses(id) on delete set null,
  date_delivered date,
  give_to text,
  remarks text,
  source_ongoing_id bigint,
  created_at timestamptz not null default now(),
  updated_at timestamptz
);

alter table public.device_inventory_items
add column if not exists source_ongoing_id bigint;

alter table public.device_inventory_items
add column if not exists client_id bigint references public.clients(id) on delete set null;

create index if not exists device_inventory_items_client_id_idx
on public.device_inventory_items(client_id);

alter table public.device_inventory_items enable row level security;

drop policy if exists "public read device inventory" on public.device_inventory_items;
drop policy if exists "public manage device inventory" on public.device_inventory_items;

create policy "public read device inventory"
on public.device_inventory_items
for select
to anon, authenticated
using (true);

create policy "public manage device inventory"
on public.device_inventory_items
for all
to anon, authenticated
using (true)
with check (true);
```

This policy is open for testing. For production, replace `to anon, authenticated` with `to authenticated` and add Supabase login.
